\begin{prob}[11]
  $$
  f(y) =
  \begin{cases}
    cy, & 0 \leq y \leq 2\\
    0, & \text{elsewhere}
  \end{cases}
  $$
  \begin{enumerate}[a)]
  \item Find the value of $c$ that makes $f(y)$ a probability density function.
    \begin{align*}
      \int_{-\infty}^\infty f(y)\, dy &= \int_0^2 cy\, \mathrm{d}y\\
      &= \frac{c}{2}y^2 \Big|_0^2\\
      &= 2c
    \end{align*}
    So $c = \frac{1}{2}$ will give us a probability density function.
  \item Find $F(y)$
  \end{enumerate}
\end{prob}
